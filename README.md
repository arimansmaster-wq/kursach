// Функция для преобразования обычного текста в HTML с форматированием
function formatTextToHTML(text) {
    if (!text) return '';
    
    let html = text;
    
    html = html.replace(/Этап (\d+)\. ([^\n]+)/g, '<div class="stage"><strong>🎯 Этап $1. $2</strong></div>');
    html = html.replace(/Задание (\d+)([^\n]*)/g, '<div class="task"><strong>📌 Задание $1$2</strong></div>');
    html = html.replace(/^([А-Я][^:\n]+):/gm, '<h3>$1:</h3>');
    html = html.replace(/^(\d+\.\d+)\.?\s+([^\n]+)/gm, '<strong style="color:#2980b9">$1 $2</strong>');
    
    let lines = html.split('\n');
    let inList = false;
    let result = [];
    for (let i = 0; i < lines.length; i++) {
        let line = lines[i];
        if (line.trim().startsWith('•') || line.match(/^\d+\./)) {
            if (!inList) {
                result.push('<ul>');
                inList = true;
            }
            result.push('<li>' + line.replace(/^[•\d+\.\s]+/, '').trim() + '</li>');
        } else {
            if (inList) {
                result.push('</ul>');
                inList = false;
            }
            if (line.trim()) {
                result.push('<p>' + line + '</p>');
            } else {
                result.push('<br>');
            }
        }
    }
    if (inList) result.push('</ul>');
    html = result.join('\n');
    
    const boldWords = ['Сферум', 'Яндекс Диск', 'Яндекс Документы', 'МЭШ', 'Моя Доска', 'VK Звонки', 'Яндекс Телемост', 'ВРМ', 'виртуальное рабочее место'];
    boldWords.forEach(word => {
        const regex = new RegExp(`\\b(${word})\\b`, 'g');
        html = html.replace(regex, '<strong>$1</strong>');
    });
    
    html = html.replace(/\n\n/g, '</p><p>');
    html = html.replace(/\n/g, '<br>');
    
    return '<div class="content-html">' + html + '</div>';
}

// Функция для создания HTML видеоплеера с поддержкой любых форматов (MP4, MOV, AVI, WEBM, MKV)
function getVideoHTML(videoFileName, description = 'Видеоматериал по теме') {
    if (!videoFileName) {
        return '<div class="video-container" style="background:#f0f0f0;"><p style="color:#999;">📹 Видео: укажите имя файла</p></div>';
    }
    
    // Определяем MIME-тип по расширению файла
    let videoType = 'video/mp4';
    const ext = videoFileName.split('.').pop().toLowerCase();
    
    switch(ext) {
        case 'mp4':
            videoType = 'video/mp4';
            break;
        case 'mov':
            videoType = 'video/quicktime';
            break;
        case 'avi':
            videoType = 'video/x-msvideo';
            break;
        case 'webm':
            videoType = 'video/webm';
            break;
        case 'mkv':
            videoType = 'video/x-matroska';
            break;
        default:
            videoType = 'video/mp4';
    }
    
    return `
        <div class="video-container">
            <video controls width="100%" preload="metadata">
                <source src="${videoFileName}" type="${videoType}">
                Ваш браузер не поддерживает тег video.
            </video>
            <p>📹 ${description}</p>
            <p style="font-size:12px; opacity:0.7;">Файл: ${videoFileName}</p>
        </div>
    `;
}

const subpointsContainer = document.getElementById("subpointsContainer");
const subButtonsDiv = document.getElementById("subButtons");
const subpointsTitle = document.getElementById("subpointsTitle");
const titleEl = document.getElementById("title");
const contentEl = document.getElementById("content");

let currentChapterIndex = 0;
let currentSubIndex = 0;

function showSubpoint(chapterIdx, subpointIdx) {
    const chapter = chapters[chapterIdx];
    const subpoint = chapter.subpoints[subpointIdx];
    titleEl.textContent = subpoint.name;
    contentEl.innerHTML = formatTextToHTML(subpoint.text);
    
    currentChapterIndex = chapterIdx;
    currentSubIndex = subpointIdx;
    
    document.querySelectorAll('.sub-btn').forEach((btn, idx) => {
        if (idx === subpointIdx) btn.classList.add('active-sub');
        else btn.classList.remove('active-sub');
    });
}

function showSubpoints(chapterIdx) {
    const chapter = chapters[chapterIdx];
    subpointsContainer.style.display = "block";
    subpointsTitle.textContent = `📌 Пункты главы «${chapter.name}»:`;
    
    subButtonsDiv.innerHTML = "";
    chapter.subpoints.forEach((sub, idx) => {
        const subBtn = document.createElement("button");
        subBtn.textContent = sub.name;
        subBtn.className = "sub-btn";
        subBtn.onclick = () => {
            showSubpoint(chapterIdx, idx);
            document.querySelectorAll('#buttons button').forEach((btn, i) => {
                if (i === chapterIdx) btn.classList.add('active');
                else btn.classList.remove('active');
            });
        };
        subButtonsDiv.appendChild(subBtn);
    });
    
    if (chapter.subpoints.length > 0) showSubpoint(chapterIdx, 0);
}

chapters.forEach((chapter, idx) => {
    const btn = document.createElement("button");
    btn.textContent = chapter.name;
    btn.onclick = () => {
        showSubpoints(idx);
        document.querySelectorAll('#buttons button').forEach((b, i) => {
            if (i === idx) b.classList.add('active');
            else b.classList.remove('active');
        });
    };
    buttonsDiv.appendChild(btn);
});

if (chapters.length > 0) {
    showSubpoints(0);
    document.querySelector('#buttons button')?.classList.add('active');
}
